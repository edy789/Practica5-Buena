DROP TABLE Extras;
DROP TABLE Turismos;
DROP TABLE Camiones;
DROP TABLE Usuarios;
DROP TABLE Vehiculos;

CREATE TABLE Vehiculos(
matricula varchar2(10) Primary key,
marca varchar2(15) NOT NULL,
modelo varchar2(15) NOT NULL,
color varchar2(15) NOT NULL,
precio double NOT NULL
);


CREATE TABLE Camiones(
matricula varchar2(10) Primary key,
marca varchar2(15) NOT NULL,
modelo varchar2(15) NOT NULL,
color varchar2(15) NOT NULL,
capacidad varchar2(15) NOT NULL,
precio double NOT NULL
FOREIGN KEY (matricula) REFERENCES Vehiculos);
);

CREATE TABLE Extras(
identificador SMALLINT PRIMARY KEY,
descripcion varchar2(50) NOT NULL
);

CREATE TABLE Turismos(
matricula varchar2(10) Primary key,
marca varchar2(15) NOT NULL,
modelo varchar2(15) NOT NULL,
color varchar2(50) NOT NULL,
numpuertas SMALLINT NOT NULL,
precio double NOT NULL,
extra SMALLINT NOT NULL, 
FOREIGN KEY (matricula) REFERENCES Vehiculos);
FOREIGN KEY(extra) REFERENCES Extras ON DELETE SET NULL
);

CREATE TABLE Usuarios(
loggin varchar2(50) Primary key,
password varchar2(50) NOT NULL
);
